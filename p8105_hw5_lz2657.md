df = df%>%
  separate(value, into = c("arm", "number"), sep = "_") %>%
  mutate(number = substr(number, 1, 2)) %>%
  unnest %>%
  gather(key = week, value = data, week_1:week_8) %>%
  mutate(week = substr(week, 6, 6)) %>%
  select(arm, number, week, data)
